79  int evaluate(const double &t, const std::vector<T> &state, std::vector<T> &dstate) const
80  {
81  //sanity checks
82  if(t<0)
83  smartmath_throw(m_name+": negative time supplied in evaluation of the dynamical system");
84  if(state.size()!=7)
85  smartmath_throw(m_name+": the state dimension needs to be 7");
86 
87  dstate.clear();
88 
89  //constant parameters
90  double radius_earth = 6378.0*pow(10,3) / m_r_scale;
91  double mu_earth = 398600.4415*pow(10,9) / (pow(m_r_scale,3)/pow(m_t_scale,2));
92  double omega_earth = 7.2921150*pow(10,-5) * m_t_scale;
93  double H0_atmosphere = 900000.0 / m_r_scale;
94 
95  //precomputations
96  T r = sqrt(state[0]*state[0]+state[1]*state[1]+state[2]*state[2]);
97  T tmp_3D = mu_earth/pow(r,3);
98 
99  //atmospheric model
100  T rho = m_param[4]*exp(-(r-radius_earth-H0_atmosphere)/m_param[5]);
101 
102  //relative velocity
103  T rel_v_x = state[3]-omega_earth*state[1];
104  T rel_v_y = state[4]+omega_earth*state[0];
105  T mod_rel_v = sqrt(rel_v_x*rel_v_x+rel_v_y*rel_v_y+state[5]*state[5]);
106 
107  //drag computation
108  T tmp_drag = 0.5*rho*m_param[6]*mod_rel_v/state[6];
109 
110  dstate.push_back(state[3]); //dx/dt
111  dstate.push_back(state[4]); //dy/dt
112  dstate.push_back(state[5]); //dz/dt
113  dstate.push_back(-tmp_3D*state[0]+m_param[0]/state[6]+m_param[7]-tmp_drag*rel_v_x); //dv_x/dt
114  dstate.push_back(-tmp_3D*state[1]+m_param[1]/state[6]+m_param[8]-tmp_drag*rel_v_y); //dv_y/dt
115  dstate.push_back(-tmp_3D*state[2]+m_param[2]/state[6]+m_param[9]-tmp_drag*state[5]); //dv_z/dt
116  dstate.push_back(-m_param[3]*sqrt(m_param[0]*m_param[0]+m_param[1]*m_param[1]+m_param[2]*m_param[2])); //dm/dt
117 
118  return 0;
119 
120  }
