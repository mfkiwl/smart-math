41  int evaluate(const double &E, const std::vector<T> &state, std::vector<T> &dstate) const{
42  //sanity checks
43  if(state.size()!=4)
44  smartmath_throw(m_name+": the state dimension needs to be 4");
45 
46  dstate.clear();
47 
48  double omega_earth = 7.2921150e-5;// * m_t_scale;
49  double radius_earth = 6378.137e3;
50  double h0 = 120.0e3/radius_earth;
51  double ht = 1000.0e3/radius_earth;
52  double mu=398600.4415e9/pow(radius_earth,3);
53 
54  T a = state[0];
55  T e = state[1];
56  T i = state[2];
57  T w = state[3];
58 
59  T delta = m_param[0]; //CD*S/m!!
60  T rho_0 = 2.438e-8*m_param[1];
61  T rho_t = 3.019e-15*m_param[2];
62 
63  T H = (h0-ht)/log(rho_t/rho_0);
64 
65  T Fminus = 1.0-e*cos(E);
66  T Fplus = 2.0-Fminus;
67  T eta = sqrt(1.0-e*e);
68  T eta2 = 1.0-e*e;
69  T fcos = (cos(E)-e)/Fminus;
70  T fsin = (eta*sin(E))/Fminus;
71  T apow=pow(a,1.5);
72  T tau = omega_earth* apow* eta * cos(i) / sqrt(mu);
73 
74  T B = 1.0 - tau * Fminus / Fplus ;
75 
76  T inter =sqrt(Fminus/Fplus);
77 
78  T Sa = B*B*Fplus/inter;
79  T Se = eta2*cos(E)*Sa/Fplus + 0.5*e*tau*B*Fminus*inter * sin(E)*sin(E);
80  T Zcos = cos(w)*fcos - sin(w)*fsin;
81  T Zsin = cos(w)*fsin + sin(w)*fcos;
82  T Si = B*Fminus*Fminus*Fplus*inter*Zcos *Zcos;
83  T Sw = sin(E)*(eta*Sa/Fplus-(e+cos(E))*(0.5*e*tau/eta)*inter*B*Fminus)-0.5*e*tau*(1.0/eta2)*sin(i)*B*Fminus*Fminus*Fplus*inter*Zcos*Zsin;
84 
85  T h = a*Fminus-1.0; // a*Fminus-(1.0-(1.0/298.0)*sin(i)*sin(i)*Zsin*Zsin);
86  T factor = a*radius_earth*delta* rho_0 * exp(-(h-h0)/H);
87 
88  dstate.push_back( - a * factor * Sa );
89  dstate.push_back( - factor * Se );
90  dstate.push_back( - factor * omega_earth * apow * sin(i) * Si / (2.0*sqrt(mu)*eta) );
91  dstate.push_back( - factor * Sw / e );
92 
93  return 0;
94  }
95

98 
99  };
