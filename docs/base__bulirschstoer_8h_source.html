47  base_bulirschstoer(const dynamics::base_dynamics<T> *dyn, const int &extrapol = 7) : base_integrator<T>("Bulirsch-Stoer algorithm", dyn), m_extrapol(extrapol){
48 
49  /* sanity checks */
50  if(m_extrapol < 1)
51  smartmath_throw("BASE_BULIRSCHSTOER: number of extrapolations needs to be non negative");
52 
53  std::vector<int> orders(m_extrapol);
54  orders[0] = 2;
55  if(m_extrapol > 1)

107  int integrate(const double &ti, const double &tend, const int &nsteps, const std::vector<T> &x0, std::vector<std::vector<T> > &x_history, std::vector<double> &t_history) const{

111 
112  std::vector<T> x(x0),xp(x0);
113  std::vector<std::vector<T> > f;
114  double t=ti, H = (tend-ti)/nsteps;
115 
116  for(int k=0; k<nsteps; k++){
117 
118  integration_step(t,H,x,xp);
119 
120  /* Saving states */
121  t+=H;
122  x=xp;
123  t_history.push_back(t);

141 
142  /* sanity checks */
143  if(n < 2)

145 
146  std::vector<T> f = y, u0 = y, u1 = y;
147  double h = H / double(n), h2 = 2.0 * h;
148 
149  m_dyn->evaluate(t, y, f);
150  for(unsigned int j = 0; j < y.size(); j++)
151  u1[j] += h * f[j];
152 
153  std::vector<T> u2 = u0;
154  m_dyn->evaluate(t + h, u1, f);
155  for(unsigned int j = 0; j < y.size(); j++)
156  u2[j] += h2 * f[j];
157 
158  std::vector<T> v = y, w = y;
159  for(int i = 2; i < n; i++)
160  {
161  v = u2;
162  u2 = u1;
163  m_dyn->evaluate(t + i * h, v, f);
164  for(unsigned int j = 0; j < y.size(); j++)
165  u2[j] += h2 * f[j];
166  w = u1;
167  u1 = v;
168  u0 = w;

203

206 
207  midpoint(y, m_orders[i-1], H, t, eta);
208  M.clear();
209  M.push_back(eta);
210 
211  if(i > 1)
212  {
213  std::vector<std::vector<T> > Mp;
214  extrapolation(i - 1, H, y, t, Mp); // recursive call
215 
216  for(int j = 1; j < i; j++)
217  {
218  eta = M[j - 1];
219  M.push_back(eta);
220  aux1 = double(m_orders[i-1]) / double(m_orders[i-1-j]);
221  aux2 = aux1 * aux1 - 1.0;
222  for(unsigned int k = 0; k < y.size(); k++)
223  M[j][k] += (eta[k] - Mp[j-1][k]) / aux2;
