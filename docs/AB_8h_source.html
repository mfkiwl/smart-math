93  int integration_step(const double &t, const int &m, const double &h, const std::vector<T> &x0, std::vector<std::vector<T> > &f, std::vector<T> &xfinal) const{
94 
95  if(f.size()!= static_cast<unsigned int>(m))

122  int initialize(const int &m, const double &ti, const double &h, const std::vector<T> &x0, std::vector<std::vector<T> > &f) const{
123 
124  f.clear();
125 
126  std::vector<T> dx(x0), x(x0), xp(x0);
127  std::vector< std::vector<T> > fp;
128 
129  /* Computing the initial saved steps */
130  m_dyn->evaluate(ti,x,dx);
131  fp.push_back(dx);
132  double t=ti;
133  for(int j=0; j<m-1; j++)
134  {
135  m_initializer->integration_step(t,-h,x,xp);
136  t-=h;
137  x=xp;
138  m_dyn->evaluate(t,x,dx);
139  fp.push_back(dx);
140  }
141 
142  /* Putting the saved steps in the right order */
143  for(int j=0; j<m; j++)
144  f.push_back(fp[m-j-1]);
145 
146  return 0;
147  }
