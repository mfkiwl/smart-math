95  int integrate(const double &ti, double &tend, const int &nsteps, const std::vector<T> &x0, std::vector<std::vector<T> > &x_history, std::vector<double> &t_history, std::vector<int> (*g)(std::vector<T> x, double d)) const{

144  }

154  factor=pow(m_tol/value,1.0/(double(m)+1.0));
155  if(value>m_tol){ // unsucessful step
156  if(m<m_order_max){
157  mold=m;
158  m++;
159  //std::cout << "increasing order" << std::endl;
160  }
161  else{
162  h*=0.9*factor;
163  m_initializer->initialize(m_order_max,t,h,x,f_max);
164  //std::cout << "decreasing time-step" << std::endl;
165  }
166  }
167  else{ // sucessful step
168  /* Checking for the events */
169  if(m_event_list.size()==0)
170  {
171  events2=g(xp,t+h);
172  }
173  else{
174  for(unsigned int index = 0; index < q; ++index)
175  {
176  events2[index] = m_event_list[index]->evaluate(t+h, xp);
177  }
178  }
179  i=0;
180  check=0;
181  while(i<q){
182  if(events2[i]-events[i]!=0){
183  check=1;
184  i=q;
185  }
186  else{
187  i++;
188  }
189  }
190  if(check==1){
191  if(sqrt(h*h)>m_minstep_events){
192  h*=0.5;
193  m_initializer->initialize(m_order_max,t,h,x,f_max);
194  //std::cout << "decreasing time-step for event detection" << std::endl;
195  }
196  else{
197  tend=t+h; // saving the termination time
198  t=tend; // trick to get out of the while loop
199  x_history.push_back(xp);
200  t_history.push_back(t);
201  if(this->m_comments)
202  std::cout << "Propagation interrupted by terminal event at time " << tend << " after " << k << " steps" << std::endl;
203  }
204  }
205  else{
206  k++; // counting the number of steps
207  x=xp; // updating state
208  t+=h; // updating current time
209  m_dyn->evaluate(t, x, dx);
210  f_max[m_order_max-1]=dx;
211  events=events2;
212  x_history.push_back(x);
213  t_history.push_back(t);
214  if(m_event_list.size()!=0)
215  {
216  for(unsigned int index = 0; index < q; ++index)
217  {
218  if(events2[index]-events[index]!=0)
219  m_event_list[index]->switch_trigger_on(t_history.back());
220  }
221  }
222  /* Step-size and order control */
223  if((m>m_order_min)&&(mold!=m-1)){
224  mold=m;
225  m--;
226  //std::cout << "decreasing order" << std::endl;
227  }
228  else if((m==m_order_min)&&(factor>=2.0)){
229  if(factor>m_multiplier){
230  factor=m_multiplier;
231  }
232  h*=factor; // updating step-size
233  m_initializer->initialize(m_order_max,t,h,x,f_max);
234  //std::cout << "increasing time-step" << std::endl;
235  }
236  else{
237  //std::cout << "keeping both time-step and order constant" << std::endl;
238  }
239  //std::cout << k << std::endl;
240  }
241  }
242  }
243 
244  return 0;
245  }

261  int integration_step(const double &t, const int &m, const double &h, const std::vector<T> &x0, const std::vector<std::vector<T> > &f, std::vector<T> &xfinal, T &er) const{

263  if(f.size()!=m)
264  smartmath_throw("INTEGRATION_STEP: wrong number of saved states for multistep integration");
265 
266  std::vector<T> x(x0), dx(x0);
267 
268  half_step(m,h,x0,f,m_gamma_Bashforth,x); // prediction
269 
270  std::vector<std::vector<T> > fp=f;
271  for(int j=0; j<m-1; j++){
272  fp[j]=f[j+1];
273  }
274  m_dyn->evaluate(t+h, x, dx);
275  fp[m-1]=dx;
276 
277  half_step(m,h,x0,fp,m_gamma_Moulton,xfinal); // correction
278

280  er=0.0;
281  for(unsigned int i=0; i<l; i++)
282  er+=pow(xfinal[i]-x[i],2);
283  er=sqrt(er);
