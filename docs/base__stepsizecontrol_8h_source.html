112  int integrate(const double &ti, double &tend, const int &nsteps, const std::vector<T> &x0, std::vector<std::vector<T> > &x_history, std::vector<double> &t_history, std::vector<int> (*g)(std::vector<T> x, double d)) const{
113 
114  x_history.clear();
115  t_history.clear();
116 
117  int k;
118  int check=0;
119  std::vector<T> x(x0), xtemp(x0);
120  std::vector<std::vector<T> > f;
121 
122  double factor=1.0, value=0.0, t=ti, h = (tend-ti)/nsteps;
123  T er=0.0*x0[0];
124 
125  std::vector<int> events, events2;
126  if(m_event_list.size()==0)
127  events=g(x0,ti);
128  else
129  {
130  events=std::vector<int>(m_event_list.size(),0);
131  for(unsigned int index = 0; index < m_event_list.size(); ++index)
132  events[index] = m_event_list[index]->evaluate(ti, x0);
133  }
134  events2=events;
135  int m=events.size();
136 
137  int i=0;
138  while(sqrt(pow(t-ti,2))<sqrt(pow(tend-ti,2))){
139 
140  if((h*h>m_maxstep_events*m_maxstep_events)&&(m_maxstep_events>0.0))
141  {
142  if(h>0.0)
143  h=m_maxstep_events;
144  else
145  h=-m_maxstep_events;
146  }
147 
148  if(sqrt(pow(tend-t,2))<sqrt(h*h))
149  h=tend-t;
150 
151  integration_step(t,m_control,h,x,f,xtemp,er);
152 
153  /* Step-size control */
154  value = evaluate_squarerootintegrationerror(er);
155  factor=pow(m_tol/value,1.0/(m_control+1.0));
156  if(er>m_tol) // unsucessful step
157  h*=0.9*factor;
158  else
159  { // sucessful step
160  /* Checking for the events */
161  if(m_event_list.size()==0)
162  events2=g(xtemp,t+h);
163  else
164  {
165  for(unsigned int index = 0; index < m; ++index)
166  events2[index] = m_event_list[index]->evaluate(t+h, xtemp);
167  }
168  k=0;
169  check=0;
170  while(k<m)
171  {
172  if(events2[k]-events[k]!=0)
173  {
174  check=1;
175  k=m;
176  }
177  else
178  k++;
179  }
180 
181  if(check==1)
182  {
183  if(sqrt(h*h)>m_minstep_events)
184  h*=0.5;
185  else
186  {
187  tend=t+h; // saving the termination time
188  t=tend; // trick to get out of the while loop
189  x_history.push_back(xtemp);
190  t_history.push_back(t);
191  if(m_event_list.size()!=0)
192  {
193  for(unsigned int index = 0; index < m; ++index)
194  {
195  if(events2[index]-events[index]!=0)
196  m_event_list[index]->switch_trigger_on(t_history.back());
197  }
198  }
199  if(this->m_comments)
200  std::cout << "Propagation interrupted by terminal event at time " << tend << " after " << i << " steps" << std::endl;
201  }
202  }
203  else
204  {
205  x=xtemp; // updating state
206  t+=h; // updating current time
207  events=events2;
208  x_history.push_back(x);
209  t_history.push_back(t);
210  /* Step-size control */
211  if(factor>m_multiplier)
212  factor=m_multiplier;
213  h*=factor; // updating step-size
214  i++; // counting the number of steps
215  }
216 
217  }
218  }
219 
220  return 0;
221  }
