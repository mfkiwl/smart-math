59  int integration_step(const double &ti, const double &tau, const std::vector<T> &x0, std::vector<T> &xfinal) const{
60 
61  /* sanity checks */
62  if(x0.size() != 2 * m_ham->get_dim())
63  smartmath_throw("INTEGRATION_STEP: state vector must have consistent dimension with Hamiltonian system");
64 
65  /* reconstituting the initial canonical variables from the state vector */
66  std::vector<T> q0, p0;
67  int n = m_ham->get_dim();
68  for(int i = 0; i < n; i++)
69  {
70  q0.push_back(x0[i]);
71  p0.push_back(x0[i + n]);
72  }
73  std::vector<T> q = q0, p = p0, dq = q0, dp = p0;
74 
75  /* performing the integration step per se using the precomputed coefficients */
76  for(int j = 0; j < m_stages; j++)
77  {
78  m_ham->DHp(ti, q0, p0, dp);
79  for(int i = 0; i < n; i++)
80  q[i] += m_c[j] * tau * dp[i];
81  m_ham->DHq(ti, q, p0, dq);
82  for(int i = 0; i < n; i++)
83  p[i] -= m_d[j] * tau * dq[i];
84  q0 = q;
85  p0 = p;
86  }
87 
88  /* reconstituting the final state vector from the propagated canonical variables */
89  xfinal.clear();
90  for(int i = 0; i < n; i++)
91  xfinal.push_back(q[i]);
92  for(int i = 0; i < n; i++)
93  xfinal.push_back(p[i]);
94 
95  return 0;
96  }
