81  int integration_step(const double &ti, const double &h, const std::vector<T> &x0, std::vector<T> &xfinal) const{
82 
83  std::vector<T> x_temp = x0, k = x0;
84  unsigned int l = x0.size();
85 
86  xfinal = x0;
87 
88  for(int i = 0; i < m_stages; i++)
89  {
90  x_temp = x0;
91  if(i == 0)
92  {
93  m_dyn->evaluate(ti + h * m_coeT[i], x_temp, k);
94  for(unsigned int j = 0; j < l; j++)
95  xfinal[j] += m_coeX[i] * k[j] * h;
96  }
97  else
98  {
99  for(unsigned int j = 0; j < l; j++)
100  x_temp[j] += m_coeK[i - 1] * k[j] * h;
101  m_dyn->evaluate(ti + h * m_coeT[i], x_temp, k);
102  for(unsigned int j = 0; j < l; j++)
103  xfinal[j] += m_coeX[i] * k[j] * h;
104  }
105  }
106 
107  return 0;
108  }
