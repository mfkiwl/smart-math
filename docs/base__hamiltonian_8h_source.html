85  int evaluate(const double &t, const std::vector<T> &state, std::vector<T> &dstate) const{
86 
87  /* sanity checks */
88  if(state.size() != 2 * m_dim)
89  smartmath_throw("EVALUATE: the Hamiltonian state must have a consistent dimension");
90  if(state.size() != dstate.size())
91  smartmath_throw("EVALUATE: the derivative of the state must have same length");
92 
93  dstate.clear();
94  /* reconstituting the canonical variables q and p from the state vector */
95  std::vector<T> dHp, dHq, q, p;
96  for(int k = 0; k < m_dim; k++)
97  {
98  q.push_back(state[k]);
99  p.push_back(state[k + m_dim]);
100  }
101  /* computing the partial derivatives of the Hamiltonian w.r.t. q and p */
102  DHq(t, q, p, dHq);
103  DHp(t, q, p, dHp);
104 
105  /* reconstituting the state derivative */
106  for(int k = 0; k < m_dim; k++)
107  dstate.push_back(dHp[k]);
108  for(int k = 0; k < m_dim; k++)
109  dstate.push_back(-dHq[k]);
110 
111  return 0;
112  }
